  function costPerUnit(recipeId,memo={}){
    if(memo[recipeId]!=null)return memo[recipeId];
    const r=recipes[recipeId];if(!r)return 0;
    const cost=(r.ingredients.reduce((s,ing)=>s+(ing.type==='buy'?ing.qty*getPrice(ing.name):ing.qty*costPerUnit(ing.recipe,memo)),0)+r.energy*energyPrice)/r.output;
    return memo[recipeId]=cost;
  }

  function expandRecipe(recipeId,needed,sections,shopping){
    const recipe=recipes[recipeId];if(!recipe)return;
    const crafts=Math.ceil(needed/recipe.output);
    if(!sections.has(recipeId))sections.set(recipeId,{recipeId,needed:0,crafts:0,produced:0,rows:new Map(),energy:0});
    const sec=sections.get(recipeId);
    sec.needed+=needed;sec.crafts+=crafts;sec.produced+=crafts*recipe.output;sec.energy+=crafts*recipe.energy;
    for(const ing of recipe.ingredients){
      const totalQty=crafts*ing.qty;
      if(ing.type==='buy'){
        const price=getPrice(ing.name);
        if(!shopping.has(ing.name))shopping.set(ing.name,{name:ing.name,qty:0,price:getPrice(ing.name)});
        shopping.get(ing.name).qty+=totalQty;shopping.get(ing.name).price=price;
        const key='buy:'+ing.name;
        if(!sec.rows.has(key))sec.rows.set(key,{name:ing.name,type:'Покупается',qty:0,price,total:0,buyName:ing.name});
        const row=sec.rows.get(key);row.qty+=totalQty;row.total+=totalQty*price;row.price=price;
      } else {
        const unit=costPerUnit(ing.recipe);const key='craft:'+ing.recipe;
        if(!sec.rows.has(key))sec.rows.set(key,{name:ing.name,type:'Крафтится',qty:0,price:unit,total:0});
        const row=sec.rows.get(key);row.qty+=totalQty;row.total+=totalQty*unit;row.price=unit;
        expandRecipe(ing.recipe,totalQty,sections,shopping);
      }
    }
    const eKey='energy';
    if(!sec.rows.has(eKey))sec.rows.set(eKey,{name:'Энергия',type:'Расход',qty:0,price:energyPrice,total:0});
    const er=sec.rows.get(eKey);er.qty+=crafts*recipe.energy;er.total+=crafts*recipe.energy*energyPrice;
  }
